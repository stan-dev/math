experimental::awaitable_operators::operator|| (2 of 6 overloads)

Wait for one operation to succeed. 

template<
    typename U,
    typename Executor>
awaitable< std::variant< std::monostate, U >, Executor > operator||(
    awaitable< void, Executor > t,
    awaitable< U, Executor > u);
