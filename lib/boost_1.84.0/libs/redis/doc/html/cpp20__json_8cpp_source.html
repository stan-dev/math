51auto co_main(config cfg) -> asio::awaitable<void>
52{
53 auto ex = co_await asio::this_coro::executor;
54 auto conn = std::make_shared<connection>(ex);
55 conn->async_run(cfg, {}, asio::consign(asio::detached, conn));
56
57 // user object that will be stored in Redis in json format.
58 user const u{"Joao", "58", "Brazil"};
59
60 // Stores and retrieves in the same request.
61 request req;
62 req.push("SET", "json-key", u); // Stores in Redis.
63 req.push("GET", "json-key"); // Retrieves from Redis.
64
65 response<ignore_t, user> resp;
66
67 co_await conn->async_exec(req, resp, asio::deferred);
68 conn->cancel();
69
70 // Prints the first ping
71 std::cout
72 << "Name: " << std::get<1>(resp).value().name << "\n"
73 << "Age: " << std::get<1>(resp).value().age << "\n"
74 << "Country: " << std::get<1>(resp).value().country << "\n";
75}
