26 if (M.size() == 0) {
27 return var(0.0);
28 }

32 matrix_d M_d = arena_M.val();
33 auto M_ldlt = M_d.ldlt();
34 if (M_ldlt.info() != Eigen::Success) {
35 constexpr double y = 0;
36 throw_domain_error("log_determinant_spd", "matrix argument", y,
37 "failed LDLT factorization");
38 }
39 // compute the inverse of A (needed for the derivative)
40 M_d.setIdentity(M.rows(), M.cols());
41 M_ldlt.solveInPlace(M_d);
42 auto arena_M_inv_transpose = to_arena(M_d.transpose());
43
44 if (M_ldlt.isNegative() || (M_ldlt.vectorD().array() <= 1e-16).any()) {
45 constexpr double y = 0;
46 throw_domain_error("log_determinant_spd", "matrix argument", y,
47 "matrix is negative definite");
48 }
49
50 var log_det = sum(log(M_ldlt.vectorD()));
51
52 reverse_pass_callback([arena_M, log_det, arena_M_inv_transpose]() mutable {
53 arena_M.adj() += log_det.adj() * arena_M_inv_transpose;
54 });
