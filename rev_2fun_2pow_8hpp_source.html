74template <typename Scal1, typename Scal2,
75 require_any_var_t<base_type_t<Scal1>, base_type_t<Scal2>>* = nullptr,
76 require_all_stan_scalar_t<Scal1, Scal2>* = nullptr>
77inline auto pow(const Scal1& base, const Scal2& exponent) {
78 if constexpr (is_complex<Scal1>::value || is_complex<Scal2>::value) {
79 return internal::complex_pow(base, exponent);
80 } else {
81 if constexpr (is_constant<Scal2>::value) {
82 if (exponent == 0.5) {
83 return sqrt(base);
84 } else if (exponent == 1.0) {
85 return base;
86 } else if (exponent == 2.0) {
87 return square(base);
88 } else if (exponent == -2.0) {
89 return inv_square(base);
90 } else if (exponent == -1.0) {
91 return inv(base);
92 } else if (exponent == -0.5) {
93 return inv_sqrt(base);
94 }
95 }
96 return make_callback_var(std::pow(value_of(base), value_of(exponent)),
97 [base, exponent](auto&& vi) mutable {
98 if (value_of(base) == 0.0) {
99 return; // partials zero, avoids 0 & log(0)
100 }
101 const double vi_mul = vi.adj() * vi.val();
102
103 if (!is_constant<Scal1>::value) {
104 forward_as<var>(base).adj()
105 += vi_mul * value_of(exponent)
106 / value_of(base);
107 }
108 if (!is_constant<Scal2>::value) {
109 forward_as<var>(exponent).adj()
110 += vi_mul * std::log(value_of(base));
111 }
112 });
113 }
114}

129template <typename Mat1, typename Mat2,
130 require_all_st_var_or_arithmetic<Mat1, Mat2>* = nullptr,
131 require_any_matrix_st<is_var, Mat1, Mat2>* = nullptr,
132 require_all_not_stan_scalar_t<Mat1, Mat2>* = nullptr>
133inline auto pow(const Mat1& base, const Mat2& exponent) {
134 check_consistent_sizes("pow", "base", base, "exponent", exponent);
135 using expr_type = decltype(as_array_or_scalar(value_of(base))
136 .pow(as_array_or_scalar(value_of(exponent))));
137 using val_type = std::conditional_t<
138 math::disjunction<is_eigen_array<Mat1>, is_eigen_array<Mat2>>::value,
139 decltype(std::declval<expr_type>().eval()),
140 decltype(std::declval<expr_type>().matrix().eval())>;
141 using ret_type = return_var_matrix_t<val_type, Mat1, Mat2>;
142 using base_t = decltype(as_array_or_scalar(base));
143 using exp_t = decltype(as_array_or_scalar(exponent));
144 using base_arena_t = arena_t<base_t>;
145 using exp_arena_t = arena_t<exp_t>;
146
147 base_arena_t arena_base = as_array_or_scalar(base);
148 exp_arena_t arena_exponent = as_array_or_scalar(exponent);
149 arena_t<ret_type> ret
150 = value_of(arena_base).pow(value_of(arena_exponent)).matrix();
151
152 reverse_pass_callback([arena_base, arena_exponent, ret]() mutable {
153 const auto& are_vals_zero = to_ref(value_of(arena_base) != 0.0);
154 const auto& ret_mul = to_ref(ret.adj().array() * ret.val().array());
155 if (!is_constant<Mat1>::value) {
156 using base_var_arena_t = arena_t<promote_scalar_t<var, base_arena_t>>;
157 forward_as<base_var_arena_t>(arena_base).adj()
158 += (are_vals_zero)
159 .select(
160 ret_mul * value_of(arena_exponent) / value_of(arena_base),
161 0);
162 }
163 if (!is_constant<Mat2>::value) {
164 using exp_var_arena_t = arena_t<promote_scalar_t<var, exp_arena_t>>;
165 forward_as<exp_var_arena_t>(arena_exponent).adj()
166 += (are_vals_zero).select(ret_mul * value_of(arena_base).log(), 0);
167 }
168 });
169 return ret_type(ret);
170}

195 } else if (exponent == 2.0) {
196 return ret_type(square(base));

198 return ret_type(inv_square(base));
199 } else if (exponent == -1.0) {
200 return ret_type(inv(base));
