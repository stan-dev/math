59 check_size_match(function, "Rows of random variable", W.rows(),

63 T_nu_ref nu_ref = nu;
64 T_S_ref S_ref = S;

79 if (include_summand<propto, T_dof>::value) {

81 }
82

84 lp -= lmgamma(k, 0.5 * nu_ref);

97 }
98 return lp;
